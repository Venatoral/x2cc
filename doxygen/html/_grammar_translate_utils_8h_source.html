158 inline typename DemoTranslator::TranslationType DemoTranslator::translate(
159  StreamConvertor& input,SemanticAction &action) const
160 {
161  const LR1Gramma::GotoInfoType & GOTO=std::get<2>(closuresInfo);
162  std::stack<int> ss;
163  ss.push(endingSym);
164  ss.push(initState);
165  while(!input.eof())
166  {
167  std::pair<int,int> gokey={ss.top(),input.peek()};
168 // std::cout << "key is "<<gokey.first << ","<<grammar.gsyms.getString(gokey.second) <<std::endl;
169  int type=grammar.gsyms.getSymbolType(gokey.second);
170 // std::cout << "input type is "<<grammar.gsyms.TYPES_INFO.getT1(type)<<std::endl;
171  if(type==GrammaSymbols::TYPE_TERM)
172  {
173  auto actionit=analyzeTable.find(gokey);
174  if(actionit!=analyzeTable.end())
175  {
176  int actiontype=std::get<2>(actionit->second);
177  if(actiontype==ACTION_TYPE::ACTION_SHIFT_IN)
178  {
179 // std::cout << "\t\tSHIFT " << grammar.gsyms.getString(gokey.second) << " going "<<std::get<0>(actionit->second)<<std::endl;
180  ss.push(gokey.second);
181  ss.push(std::get<0>(actionit->second));
182  input.goForward();
183  }else if(actiontype==ACTION_TYPE::ACTION_REDUCE){
184  int i=std::get<0>(actionit->second),j=std::get<1>(actionit->second);
185  int len=2*grammar.getRightSentences(i).at(j).getLength();
186 // std::cout << "length is "<<len<<std::endl;
187  while(len--)
188  {
189  ss.pop();
190  }
191  std::pair<int,int> tempkey={ss.top(),i};
192  ss.push(i);
193 // std::cout << "after removed,top is <"<<tempkey.first<<","<<grammar.gsyms.getString(tempkey.second)<<">"<<std::endl;
194  auto tempit=GOTO.find(tempkey);
195  if(tempit!=GOTO.end())
196  {
197  ss.push(tempit->second);
198  }else action.onNoGotoError();
199  std::cout << "\t\treducing "<<grammar.Gramma::toString(i, j)<<std::endl;
200  action.onReduce(i,j);
201  }else if(actiontype==ACTION_TYPE::ACTION_ACCEPT){
202  std::cout << "ACCEPT" << std::endl;
203  break;
204  }
205  }else{
206  action.onNoActionError();
207  while(getchar()!='\n');
208  }
209  }else if(type==GrammaSymbols::TYPE_VAR){
210  auto goit=GOTO.find(gokey);

212  {
213 // std::cout << "GOTO " << goit->second <<std::endl;
214  ss.push(goit->second);
215  }else{
216  action.onNoGotoError();
217  }
218  }
219  }

221 }
